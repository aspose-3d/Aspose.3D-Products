Scene scene = new Scene(file);
string text =null;

//Adjon hozzá jelszót a vízjel DecodeWatermark módszerrel történő ellenőrzéséhez
try
{
    scene.RootNode.Accept((Node node) =>
    {
        var mesh = node.GetEntity<Mesh>();
        if (mesh != null)
        {
            text = Watermark.DecodeWatermark(mesh, "1234");
            if (text != null)
                return false;
            }
            return true;
    });
}
catch (UnauthorizedAccessException)
{
    return "Password error";
}

//Null értéket ad vissza, ha ennek a fájlnak nincs vízjele, ha van vízjel, adja vissza a vízjel tartalmát

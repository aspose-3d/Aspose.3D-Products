```cs

//File sumber yang perlu diberi air dan file keluaran setelah disimpan
string file = "template.x";
string output =System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".fbx";

// Membuat contoh adegan
Scene scene = new Scene(file);

//Tambahkan tanda air dan kata sandi ke file
var numMeshes = 0;
scene.RootNode.Accept((Node node) =>
{
    var mesh = node.GetEntity<Mesh>();
    if (mesh != null)
    {
        numMeshes++;
        mesh = Watermark.EncodeWatermark(mesh, "HelloWorld", "1234");
        if (mesh != null)
        {
            node.Entity = mesh;
        }
    }
    return true;
});

//Menyimpan berkas dalam format yang Anda inginkan
scene.Save(output, FileFormat.FBX7400ASCII);


```

//File sumber yang perlu ditandai air untuk verifikasi
string file = "template.usd";

// Membuat contoh adegan
Scene scene = new Scene(file);
string text =null;

//Tambahkan kata sandi untuk memverifikasi tanda air dengan metode DecodeWatermark
try
{
    scene.RootNode.Accept((Node node) =>
    {
        var mesh = node.GetEntity<Mesh>();
        if (mesh != null)
        {
            text = Watermark.DecodeWatermark(mesh, "1234");
            if (text != null)
                return false;
            }
            return true;
    });
}
catch (UnauthorizedAccessException)
{
    return "Password error";
}

//Mengembalikan nol jika file ini tidak memiliki tanda air, jika ada tanda air, kembalikan konten tanda air
return text;

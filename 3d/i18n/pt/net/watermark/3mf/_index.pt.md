```cs

//O arquivo de origem que precisa ser marcado com marca d'água e o arquivo de saída depois de salvar
string file = "template.3mf";
string output =System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".fbx";

// Criar uma instância de cena
Scene scene = new Scene(file);

//Adicionar marca d'água e senha aos arquivos
var numMeshes = 0;
scene.RootNode.Accept((Node node) =>
{
    var mesh = node.GetEntity<Mesh>();
    if (mesh != null)
    {
        numMeshes++;
        mesh = Watermark.EncodeWatermark(mesh, "HelloWorld", "1234");
        if (mesh != null)
        {
            node.Entity = mesh;
        }
    }
    return true;
});

//Salve o arquivo no formato desejado
scene.Save(output, FileFormat.FBX7400ASCII);


```

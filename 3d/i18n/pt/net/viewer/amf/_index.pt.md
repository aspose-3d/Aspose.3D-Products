```cs

string output = System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".html";

// Carregar AMF cena através de uma instância de cena
var scene = new ThreeD.Scene("template.amf");
// Criar um objeto de Html5SaveOptions e definir propriedades para formatação
var options = new ThreeD.Formats.Html5SaveOptions()
{
    // Desligue a grade
    ShowGrid = false,
    // Desligar a interface do usuário
    ShowUI = false
};

// Salvar 3D cena como HTML5
scene.Save(output, options);
// Carga resultante HTML no navegador padrão
System.Diagnostics.Process.Start(output);


```

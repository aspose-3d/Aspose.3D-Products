```cs

string output = System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".html";

// Charger RVM scène via une instance de scène
var scene = new ThreeD.Scene("template.rvm");
// Créer un objet de Html5SaveOptions et définir des propriétés pour le formatage
var options = new ThreeD.Formats.Html5SaveOptions()
{
    // Éteignez la grille
    ShowGrid = false,
    // Désactivez l'interface utilisateur
    ShowUI = false
};

// Enregistrer 3D scène en tant que HTML5
scene.Save(output, options);
// Charge résultante HTML dans le navigateur par défaut
System.Diagnostics.Process.Start(output);
```

//Исходные файлы, которые должны быть отмечены водяным знаком для проверки
string file = "template.usdz";

// Создать экземпляр сцены
Scene scene = new Scene(file);
string text =null;

//Добавить пароль для проверки водяного знака методом DecodeWatermark
try
{
    scene.RootNode.Accept((Node node) =>
    {
        var mesh = node.GetEntity<Mesh>();
        if (mesh != null)
        {
            text = Watermark.DecodeWatermark(mesh, "1234");
            if (text != null)
                return false;
            }
            return true;
    });
}
catch (UnauthorizedAccessException)
{
    return "Password error";
}

//Возвращает NULL, если этот файл не имеет водяного знака, если есть водяной знак, возвращает содержимое водяного знака
return text;

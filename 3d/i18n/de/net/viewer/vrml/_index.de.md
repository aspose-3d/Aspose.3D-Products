```cs

string output = System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".html";

// Laden Sie VRML Szene über eine Instanz von Szene
var scene = new ThreeD.Scene("template.vrml");
// Erstellen Sie ein Objekt von Html5SaveOptions und legen Sie Eigenschaften für die Formatierung fest
var options = new ThreeD.Formats.Html5SaveOptions()
{
    // Schalten Sie das Gitter aus
    ShowGrid = false,
    // Schalten Sie die Benutzer oberfläche aus
    ShowUI = false
};

// 3D Szene speichern als HTML5
scene.Save(output, options);
// Laden Sie das resultierende HTML im Standard browser
System.Diagnostics.Process.Start(output);


```

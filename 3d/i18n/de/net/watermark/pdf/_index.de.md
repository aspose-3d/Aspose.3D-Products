```cs

//Die Quelldatei, die mit Wasser markiert werden muss, und die Ausgabe datei nach dem Speichern
string file = "template.pdf";
string output =System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".fbx";

// Erstellen Sie eine Instanz der Szene
Scene scene = new Scene(file);

//Fügen Sie Wasser zeichen und Passwort zu Dateien hinzu
var numMeshes = 0;
scene.RootNode.Accept((Node node) =>
{
    var mesh = node.GetEntity<Mesh>();
    if (mesh != null)
    {
        numMeshes++;
        mesh = Watermark.EncodeWatermark(mesh, "HelloWorld", "1234");
        if (mesh != null)
        {
            node.Entity = mesh;
        }
    }
    return true;
});

//Speichern Sie die Datei im gewünschten Format
scene.Save(output, FileFormat.FBX7400ASCII);


```

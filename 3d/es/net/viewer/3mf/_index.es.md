string output = System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".html";

// cargar 3MF escena a través de una instancia de Escena
var scene = new ThreeD.Scene("template.3mf");
// crear un objeto de Html5SaveOptions y establecer propiedades para formatear
var options = new ThreeD.Formats.Html5SaveOptions()
{
    // apagar la red
    ShowGrid = false,
    // apagar la interfaz de usuario
    ShowUI = false
};

// guardar 3D escena como HTML5
scene.Save(output, options);
// cargar HTML resultante en el navegador predeterminado
System.Diagnostics.Process.Start(output);

Se creó para evitar las limitaciones y los problemas de otros formatos de archivo 3D, como STL, para trabajar con las últimas versiones de las impresoras 3D. 3MF es un formato de archivo relativamente nuevo desarrollado y publicado por el consorcio 3MF.

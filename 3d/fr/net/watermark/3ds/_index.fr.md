```cs

//Le fichier source qui doit être filigrané et le fichier de sortie après l'enregistrement
string file = "template.3ds";
string output =System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".fbx";

// créer une instance de Scene
Scene scene = new Scene(file);

//Ajouter un filigrane et un mot de passe aux fichiers
var numMeshes = 0;
scene.RootNode.Accept((Node node) =>
{
    var mesh = node.GetEntity<Mesh>();
    if (mesh != null)
    {
        numMeshes++;
        mesh = Watermark.EncodeWatermark(mesh, "HelloWorld", "1234");
        if (mesh != null)
        {
            node.Entity = mesh;
        }
    }
    return true;
});

//Enregistrez le fichier dans le format que vous voulez
scene.Save(output, FileFormat.FBX7400ASCII);


```

Autodesk 3D Studio est présent sur le marché des formats de fichiers 3D depuis les années 1990 et a maintenant évolué vers 3D Studio MAX pour travailler avec 3D la modélisation, l'animation et le rendu.

```cs
//स्रोत फ़ाइल जिसे वॉटरमार्क करने की आवश्यकता है और सहेजने के बाद आउटपुट फ़ाइल
string file = "template.gltf";
string output =System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".fbx";

// दृश्य का एक उदाहरण बनाएँ
Scene scene = new Scene(file);

//फाइलों में वॉटरमार्क और पासवर्ड जोड़ें
var numMeshes = 0;
scene.RootNode.Accept((Node node) =>
{
    var mesh = node.GetEntity<Mesh>();
    if (mesh != null)
    {
        numMeshes++;
        mesh = Watermark.EncodeWatermark(mesh, "HelloWorld", "1234");
        if (mesh != null)
        {
            node.Entity = mesh;
        }
    }
    return true;
});

//फ़ाइल को अपने इच्छित प्रारूप में सहेजें
scene.Save(output, FileFormat.FBX7400ASCII);


```
